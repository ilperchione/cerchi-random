   <script type="text/javascript">

      let counter=0

      function setup() {
        createCanvas(1000,1000);
        frameRate(60);
      }

      function draw() {
        let x=random(1000)
        let y=random(1000)
        let d=random(30,60)
        let r=random(256)
        let g=random(256)
        let b=random(256)
        let alpha=random(256)
        noStroke()
        fill(r, g, b, alpha)
        circle(x, y, d)
        if(counter>1000){
          counter=0
          background(255,255,255)
        }
        counter=counter+1
      }

   </script>
